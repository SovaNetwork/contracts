// src/components/portfolio/portfolio-summary.tsx
'use client'

import { motion } from 'framer-motion'
import { useAccount } from 'wagmi'
import { Card, CardContent } from '@/components/ui/card'
import { 
  TrendingUp, 
  TrendingDown, 
  DollarSign, 
  Percent, 
  Target,
  Zap
} from 'lucide-react'

import { useTokenBalance } from '@/hooks/web3/use-token-balance'
import { useStakingData } from '@/hooks/web3/use-staking-data'
import { CONTRACT_ADDRESSES } from '@/contracts/addresses'
import { formatTokenAmount, formatUSD } from '@/lib/utils'
import { formatUnits } from 'viem'
import { baseSepolia } from 'viem/chains'

export function PortfolioSummary() {
  const { isConnected } = useAccount()
  
  const sovaBTCBalance = useTokenBalance(CONTRACT_ADDRESSES[baseSepolia.id].SOVABTC)
  const sovaBalance = useTokenBalance(CONTRACT_ADDRESSES[baseSepolia.id].SOVA_TOKEN)
  const stakingData = useStakingData(CONTRACT_ADDRESSES[baseSepolia.id].STAKING)
  
  // Mock prices
  const btcPrice = 45000
  const sovaPrice = 0.1
  
  // Calculate values
  const sovaBTCValue = Number(sovaBTCBalance.formattedBalance) * btcPrice
  const sovaValue = Number(sovaBalance.formattedBalance) * sovaPrice
  const stakedValue = Number(formatUnits(stakingData.stakedAmount, 8)) * btcPrice
  const pendingRewardsValue = Number(formatUnits(stakingData.earnedRewards, 18)) * sovaPrice
  
  const totalValue = sovaBTCValue + sovaValue + stakedValue + pendingRewardsValue
  
  // Mock performance data
  const dayChange = 0.024 // +2.4%
  const dayChangeValue = totalValue * dayChange
  
  if (!isConnected) {
    return (
      <Card className="defi-card">
        <CardContent className="p-8 text-center">
          <p className="text-slate-400">Connect your wallet to view portfolio summary</p>
        </CardContent>
      </Card>
    )
  }

  const metrics = [
    {
      title: 'Total Portfolio Value',
      value: formatUSD(totalValue),
      change: `${dayChange >= 0 ? '+' : ''}${(dayChange * 100).toFixed(2)}%`,
      changeValue: `${dayChange >= 0 ? '+' : ''}${formatUSD(Math.abs(dayChangeValue))}`,
      changeType: dayChange >= 0 ? 'positive' : 'negative',
      icon: DollarSign,
      gradient: 'from-defi-blue-400 to-defi-purple-400',
    },
    {
      title: 'Active Positions',
      value: formatUSD(stakedValue),
      change: '+5.8%',
      changeValue: 'Staking rewards',
      changeType: 'positive',
      icon: Target,
      gradient: 'from-defi-purple-400 to-defi-pink-400',
    },
    {
      title: 'Pending Rewards',
      value: formatUSD(pendingRewardsValue),
      change: `${formatTokenAmount(stakingData.earnedRewards, 18)} SOVA`,
      changeValue: 'Claimable now',
      changeType: 'positive',
      icon: Zap,
      gradient: 'from-defi-pink-400 to-defi-blue-400',
    },
    {
      title: 'Current APY',
      value: `${stakingData.apy.toFixed(2)}%`,
      change: '+0.3%',
      changeValue: 'vs last week',
      changeType: 'positive',
      icon: Percent,
      gradient: 'from-defi-green-400 to-defi-blue-400',
    },
  ]

  return (
    <Card className="defi-card border-2 border-defi-blue-500/30 bg-gradient-to-r from-defi-blue-500/5 to-defi-purple-500/5">
      <CardContent className="p-8">
        <div className="grid gap-6 md:grid-cols-2 lg:grid-cols-4">
          {metrics.map((metric, index) => (
            <motion.div
              key={metric.title}
              initial={{ opacity: 0, y: 20 }}
              animate={{ opacity: 1, y: 0 }}
              transition={{ duration: 0.5, delay: index * 0.1 }}
              className="text-center space-y-3"
            >
              <div className={`w-12 h-12 mx-auto rounded-xl bg-gradient-to-r ${metric.gradient} flex items-center justify-center`}>
                <metric.icon className="h-6 w-6 text-white" />
              </div>
              
              <div className="space-y-1">
                <p className="text-sm text-slate-400">{metric.title}</p>
                <p className="text-2xl font-bold text-white">{metric.value}</p>
                <div className="flex items-center justify-center gap-1">
                  {metric.changeType === 'positive' ? (
                    <TrendingUp className="h-3 w-3 text-defi-green-400" />
                  ) : (
                    <TrendingDown className="h-3 w-3 text-defi-red-400" />
                  )}
                  <span className={`text-xs font-medium ${
                    metric.changeType === 'positive' ? 'text-defi-green-400' : 'text-defi-red-400'
                  }`}>
                    {metric.change}
                  </span>
                </div>
                <p className="text-xs text-slate-500">{metric.changeValue}</p>
              </div>
            </motion.div>
          ))}
        </div>
      </CardContent>
    </Card>
  )
}
